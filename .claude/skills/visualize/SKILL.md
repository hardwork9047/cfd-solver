あなたは LBM-DEM シミュレーション結果の可視化専門家アシスタントです。`pvpython`（ParaView の Python バインディング）を使って VTK ファイルから多モードの画像・アニメーションを生成します。

## 実行環境メモ

- `pvpython` のパス: `/usr/bin/pvpython`
- `paraview` モジュールは自動 import できないため必ず `PYTHONPATH=/usr/lib/python3/dist-packages` を設定して実行する
- `--mesa` オプションは使用不可（SIGSEGV になる）
- 実行コマンド: `PYTHONPATH=/usr/lib/python3/dist-packages pvpython <script.py>`

## 手順

### ステップ 2: pvpython スクリプトの生成と実行

以下のスクリプトを `/tmp/pviz_<timestamp>.py` に書き出して実行する。
`PARAVIEW_DIR` と `OUTPUT_DIR` を実際のパスで置換すること。

```python
#!/usr/bin/env pvpython
"""Multi-mode visualization script — generated by /visualize skill."""
import glob, os, subprocess, math
from paraview.simple import *

PARAVIEW_DIR = "PARAVIEW_DIR_PLACEHOLDER"
OUTPUT_DIR   = "OUTPUT_DIR_PLACEHOLDER"
os.makedirs(OUTPUT_DIR, exist_ok=True)

fluid_files    = sorted(glob.glob(os.path.join(PARAVIEW_DIR, "fluid_*.vtk")))
particle_files = sorted(glob.glob(os.path.join(PARAVIEW_DIR, "particles_*.vtk")))
N = len(fluid_files)
print(f"fluid={N}, particles={len(particle_files)}")

# --- ドメイン情報を最終ステップから取得 ---
_probe = LegacyVTKReader(FileNames=[fluid_files[-1]])
UpdatePipeline()
bounds = _probe.GetDataInformation().GetBounds()
cx = (bounds[0]+bounds[1])/2
cy = (bounds[2]+bounds[3])/2
cz = (bounds[4]+bounds[5])/2
bx, by, bz = bounds[1]-bounds[0], bounds[3]-bounds[2], bounds[5]-bounds[4]
diag = math.sqrt(bx**2 + by**2 + bz**2)
Delete(_probe)

# カメラプリセット（斜め上方から見る 45° 視点）
def set_iso_camera(view):
    view.CameraPosition  = [cx + diag*0.9, cy + diag*0.6, cz + diag*0.9]
    view.CameraFocalPoint = [cx, cy, cz]
    view.CameraViewUp    = [0, 0, 1]

def new_view(size=(800,600)):
    v = CreateRenderView()
    v.ViewSize = list(size)
    v.Background = [0.08, 0.08, 0.12]
    return v

def frame_path(d, step_str):
    return os.path.join(d, f"frame_{step_str}.png")

def make_mp4(png_dir, out_mp4, fps=10):
    ret = subprocess.call([
        "ffmpeg", "-y", "-framerate", str(fps),
        "-pattern_type", "glob",
        "-i", os.path.join(png_dir, "frame_*.png"),
        "-c:v", "libx264", "-pix_fmt", "yuv420p", out_mp4
    ], stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
    return ret == 0

# ================================================================
# MODE 1: ボリュームレンダリング — 速度場を透明感のある 3D 体積で表示
#         カメラをゆっくり周回させて立体感を出す
# ================================================================
print("=== MODE 1: Volume rendering (speed) ===")
d1 = os.path.join(OUTPUT_DIR, "frames_volume"); os.makedirs(d1, exist_ok=True)
for i, ff in enumerate(fluid_files):
    v = new_view()
    fl = LegacyVTKReader(FileNames=[ff])
    disp = Show(fl, v)
    disp.Representation = 'Volume'
    ColorBy(disp, ('POINTS', 'speed'))
    disp.RescaleTransferFunctionToDataRange()
    lut = GetColorTransferFunction('speed')
    lut.ApplyPreset('Cool to Warm', True)
    # 透明度関数: 遅い部分を透明に
    otf = GetOpacityTransferFunction('speed')
    r = lut.RGBPoints[0], lut.RGBPoints[-4]  # min, max
    vmin, vmax = disp.MapScalars and 0.0 or 0.0, 0.15
    otf.Points = [vmin, 0.0, 0.5, 0.0,
                  vmax*0.3, 0.05, 0.5, 0.0,
                  vmax*0.7, 0.3, 0.5, 0.0,
                  vmax, 0.8, 0.5, 0.0]
    # カメラ周回: 360° / N フレーム
    angle = (360.0 / N) * i
    rad = math.radians(angle)
    r_cam = diag * 1.1
    v.CameraPosition  = [cx + r_cam*math.cos(rad), cy + r_cam*math.sin(rad), cz + diag*0.5]
    v.CameraFocalPoint = [cx, cy, cz]
    v.CameraViewUp    = [0, 0, 1]
    step_str = os.path.basename(ff).replace('fluid_','').replace('.vtk','')
    SaveScreenshot(frame_path(d1, step_str), v, magnification=1)
    Delete(v)
ok1 = make_mp4(d1, os.path.join(OUTPUT_DIR, "anim_volume_speed.mp4"))
print(f"  -> anim_volume_speed.mp4 {'OK' if ok1 else 'FAILED'}")

# ================================================================
# MODE 2: 粒子球立体 — 3D Glyphs（球）で粒子を描画 + 流体輪郭線
#         粒子の速度で色付け、isometric 視点固定
# ================================================================
print("=== MODE 2: Particles 3D glyphs ===")
d2 = os.path.join(OUTPUT_DIR, "frames_particles3d"); os.makedirs(d2, exist_ok=True)
pairs = list(zip(fluid_files, particle_files)) if particle_files else [(f, None) for f in fluid_files]
for ff, pf in pairs:
    v = new_view()
    # 流体輪郭（薄い外枠）
    fl = LegacyVTKReader(FileNames=[ff])
    disp_fl = Show(fl, v)
    disp_fl.Representation = 'Outline'
    disp_fl.AmbientColor = [0.3, 0.3, 0.4]
    disp_fl.DiffuseColor = [0.3, 0.3, 0.4]
    # 粒子を球として描画
    if pf and os.path.exists(pf):
        pts = LegacyVTKReader(FileNames=[pf])
        UpdatePipeline()
        glyph = Glyph(Input=pts, GlyphType='Sphere')
        glyph.ScaleArray = ['POINTS', 'radius']
        glyph.ScaleFactor = 2.0   # 格子単位での直径 = 2*radius
        glyph.GlyphMode = 'All Points'
        glyph.GlyphType.ThetaResolution = 12
        glyph.GlyphType.PhiResolution   = 12
        disp_g = Show(glyph, v)
        disp_g.Representation = 'Surface'
        # 粒子速度の x 成分で色付け
        ColorBy(disp_g, ('POINTS', 'velocity'), component=0)
        disp_g.RescaleTransferFunctionToDataRange()
        lut_p = GetColorTransferFunction('velocity_X')
        lut_p.ApplyPreset('Cool to Warm', True)
        disp_g.Specular = 0.5
        disp_g.SpecularPower = 40
    set_iso_camera(v)
    step_str = os.path.basename(ff).replace('fluid_','').replace('.vtk','')
    SaveScreenshot(frame_path(d2, step_str), v, magnification=1)
    Delete(v)
ok2 = make_mp4(d2, os.path.join(OUTPUT_DIR, "anim_particles_3d.mp4"))
print(f"  -> anim_particles_3d.mp4 {'OK' if ok2 else 'FAILED'}")

# ================================================================
# MODE 3: 3断面スライス — x/y/z 中央断面の速度を 1 画面に 3 分割表示
#         ComparativeRenderView で並べる
# ================================================================
print("=== MODE 3: 3-panel slices ===")
d3 = os.path.join(OUTPUT_DIR, "frames_slices3"); os.makedirs(d3, exist_ok=True)
normals = [[1,0,0],[0,1,0],[0,0,1]]
origins = [[cx,cy,cz],[cx,cy,cz],[cx,cy,cz]]
for ff, pf in pairs:
    # 3つのビューを横並びに手動合成 (SaveScreenshot を3回撮って imagemagick で結合)
    tmp_imgs = []
    for ni, (norm, orig) in enumerate(zip(normals, origins)):
        v = new_view(size=(400, 400))
        fl = LegacyVTKReader(FileNames=[ff])
        sl = Slice(Input=fl)
        sl.SliceType = "Plane"
        sl.SliceType.Normal = norm
        sl.SliceType.Origin = orig
        disp = Show(sl, v)
        ColorBy(disp, ('POINTS', 'speed'))
        disp.RescaleTransferFunctionToDataRange()
        GetColorTransferFunction('speed').ApplyPreset('Rainbow Desaturated', True)
        v.ResetCamera()
        # 粒子を点として重ねる
        if pf and os.path.exists(pf):
            pts = LegacyVTKReader(FileNames=[pf])
            d_pts = Show(pts, v)
            d_pts.Representation = 'Point Gaussian'
            d_pts.GaussianRadius = 0.8
            d_pts.ShaderPreset = 'Sphere'
            ColorBy(d_pts, None)
            d_pts.AmbientColor = [1.0, 0.4, 0.0]
            d_pts.DiffuseColor = [1.0, 0.4, 0.0]
        tmp_path = os.path.join(d3, f"_tmp_panel{ni}.png")
        SaveScreenshot(tmp_path, v, magnification=1)
        tmp_imgs.append(tmp_path)
        Delete(v)
    # imagemagick で横並び結合
    step_str = os.path.basename(ff).replace('fluid_','').replace('.vtk','')
    out_frame = frame_path(d3, step_str)
    ret = subprocess.call(["convert", "+append"] + tmp_imgs + [out_frame],
                          stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
    if ret != 0:
        # imagemagick がない場合は最初のパネルのみ
        import shutil; shutil.copy(tmp_imgs[0], out_frame)
    for t in tmp_imgs:
        try: os.remove(t)
        except: pass
ok3 = make_mp4(d3, os.path.join(OUTPUT_DIR, "anim_slices_3panel.mp4"))
print(f"  -> anim_slices_3panel.mp4 {'OK' if ok3 else 'FAILED'}")

# ================================================================
# MODE 4: 流線（Streamlines）— 速度ベクトルの流線を 3D 表示
# ================================================================
print("=== MODE 4: Streamlines ===")
d4 = os.path.join(OUTPUT_DIR, "frames_streamlines"); os.makedirs(d4, exist_ok=True)
for ff in fluid_files:
    v = new_view()
    fl = LegacyVTKReader(FileNames=[ff])
    # 流線: ランダム 200 点から seed
    stream = StreamTracer(Input=fl, SeedType='Point Cloud')
    stream.Vectors = ['POINTS', 'velocity']
    stream.SeedType.Center = [cx, cy, cz]
    stream.SeedType.Radius = min(bx, by, bz) * 0.45
    stream.SeedType.NumberOfPoints = 200
    stream.MaximumStreamlineLength = bx * 1.5
    disp_s = Show(stream, v)
    disp_s.Representation = 'Surface'
    ColorBy(disp_s, ('POINTS', 'speed'))
    disp_s.RescaleTransferFunctionToDataRange()
    lut_s = GetColorTransferFunction('speed')
    lut_s.ApplyPreset('Cool to Warm', True)
    # 輪郭
    fl2 = LegacyVTKReader(FileNames=[ff])
    disp_out = Show(fl2, v)
    disp_out.Representation = 'Outline'
    disp_out.AmbientColor = [0.4, 0.4, 0.5]
    set_iso_camera(v)
    step_str = os.path.basename(ff).replace('fluid_','').replace('.vtk','')
    SaveScreenshot(frame_path(d4, step_str), v, magnification=1)
    Delete(v)
ok4 = make_mp4(d4, os.path.join(OUTPUT_DIR, "anim_streamlines.mp4"))
print(f"  -> anim_streamlines.mp4 {'OK' if ok4 else 'FAILED'}")

# ================================================================
# MODE 5: 密度ボリューム + 粒子球オーバーレイ
#         密度場を半透明ボリュームで表示し、粒子を不透明球で重ねる
# ================================================================
print("=== MODE 5: Density volume + particles overlay ===")
d5 = os.path.join(OUTPUT_DIR, "frames_density_particles"); os.makedirs(d5, exist_ok=True)
for ff, pf in pairs:
    v = new_view()
    fl = LegacyVTKReader(FileNames=[ff])
    disp_v = Show(fl, v)
    disp_v.Representation = 'Volume'
    ColorBy(disp_v, ('POINTS', 'density'))
    disp_v.RescaleTransferFunctionToDataRange()
    lut_d = GetColorTransferFunction('density')
    lut_d.ApplyPreset('Blue Orange (diverging)', True)
    otf_d = GetOpacityTransferFunction('density')
    otf_d.Points = [0.99, 0.0, 0.5, 0.0,
                    1.0,  0.0, 0.5, 0.0,
                    1.005,0.05,0.5, 0.0,
                    1.01, 0.15,0.5, 0.0,
                    1.02, 0.4, 0.5, 0.0]
    # 粒子球
    if pf and os.path.exists(pf):
        pts = LegacyVTKReader(FileNames=[pf])
        UpdatePipeline()
        glyph = Glyph(Input=pts, GlyphType='Sphere')
        glyph.ScaleArray = ['POINTS', 'radius']
        glyph.ScaleFactor = 2.0
        glyph.GlyphMode = 'All Points'
        glyph.GlyphType.ThetaResolution = 10
        glyph.GlyphType.PhiResolution   = 10
        disp_g = Show(glyph, v)
        disp_g.Representation = 'Surface'
        ColorBy(disp_g, None)
        disp_g.AmbientColor = [1.0, 0.55, 0.0]
        disp_g.DiffuseColor = [1.0, 0.55, 0.0]
        disp_g.Opacity = 1.0
        disp_g.Specular = 0.6
        disp_g.SpecularPower = 50
    set_iso_camera(v)
    step_str = os.path.basename(ff).replace('fluid_','').replace('.vtk','')
    SaveScreenshot(frame_path(d5, step_str), v, magnification=1)
    Delete(v)
ok5 = make_mp4(d5, os.path.join(OUTPUT_DIR, "anim_density_particles.mp4"))
print(f"  -> anim_density_particles.mp4 {'OK' if ok5 else 'FAILED'}")

# ================================================================
# 静止画スナップショット（最終ステップ、全モード）
# ================================================================
print("=== Static snapshots (final step) ===")
# 既に各モードのフレームが生成されているので最終フレームをスナップショットとしてコピー
import shutil
last_step = os.path.basename(fluid_files[-1]).replace('fluid_','').replace('.vtk','')
for mode, d in [("volume", d1), ("particles3d", d2), ("slices3", d3), ("streamlines", d4), ("density_particles", d5)]:
    src_f = frame_path(d, last_step)
    dst_f = os.path.join(OUTPUT_DIR, f"snapshot_{mode}.png")
    if os.path.exists(src_f):
        shutil.copy(src_f, dst_f)
        print(f"  snapshot_{mode}.png")

print("\n=== DONE ===")
print(f"OUTPUT: {OUTPUT_DIR}")
results = {
    "anim_volume_speed.mp4": ok1,
    "anim_particles_3d.mp4": ok2,
    "anim_slices_3panel.mp4": ok3,
    "anim_streamlines.mp4": ok4,
    "anim_density_particles.mp4": ok5,
}
for f, ok in results.items():
    print(f"  {'✓' if ok else '✗'} {f}")
```

### 膜細孔（円柱）の滑らかな描画 — 解析的円柱メッシュ

**重要:** 膜細孔の円柱を VTK の `solid` フィールドから `Threshold` + `ExtractSurface`
で抽出すると、粗格子（例: 半径3.0 LU / 48×24×24）ではボクセルの階段状になり
**四角く見える**。物理計算は正しく円柱だが、見た目だけが粗い。

これを避けるため、ジオメトリ定義（中心 x, y + 半径）から**解析的な滑らかな円柱**を
ParaView の `Cylinder` ソースで生成して重畳する。再計算不要で完全な真円断面になる。

**ヘルパー:** `.claude/skills/visualize/cylinder_helper.py`

```python
import sys
sys.path.insert(0, ".claude/skills/visualize")
from cylinder_helper import add_analytic_cylinders

# fluid の Outline を表示した後に呼ぶ
add_analytic_cylinders(
    view,
    geometry_json="configs/lbm_dem/geometries/four_cylinder_3d.json",
    nz=24,                # ドメイン奥行き（円柱の高さ=z-span）
    opacity=0.55,         # 背後の粒子を透かす
    color=(0.25,0.55,0.95),
    resolution=48,        # 円周ファセット数（48で滑らか）
)
```

**実装上の注意（geometry3d.py の `_add_cylinder` 規約に一致させること）:**
- 円柱は **z 軸方向**（軸が z に平行）、ドメイン全深さを貫通する
- ParaView の `Cylinder` は既定で **+Y 軸**方向なので、X 軸まわりに -90° 回転して
  +Z 向きにし、中心 `(x, y, nz/2)` へ平行移動する（ヘルパー内で実施済み）
- ジオメトリ JSON からは x, y, radius のみ読む。z 範囲は引数 `nz` で与える

**旧来の `Threshold(solid)` 方式は使わないこと**（四角く見えるため）。全可視化モードで
シリンダーを描く場合は `add_analytic_cylinders()` に統一する。

## 注意事項

- **imagemagick** (`convert`) がインストールされていない場合、3断面モードはパネル1枚のみになる
- `pvpython` は毎フレーム `Delete(view)` でメモリ解放するが、フレーム数が多いと重い。フレームを間引く場合はスクリプト内で `fluid_files = fluid_files[::2]` のようにスライスする
- ボリュームレンダリングは STRUCTURED_POINTS データが必要。POLYDATA（粒子 VTK）には Volume 表示は使えない
- `Glyph` で球を描画する際の `ScaleFactor` は格子単位のため、`particle_radius=2.0` なら `ScaleFactor=2.0`（直径）が適切
